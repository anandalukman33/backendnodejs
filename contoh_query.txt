Get All User
query {
  users {
    id
    name
    email
  }
}


Add User 
mutation {
  addUser(name: "John Doe", email: "john@example.com", password: "password123") {

  }
}



Get All Blogs 
query {
  blogs {
    id
    title
    content
    metadata {
      tags
      views
      author
      rating
      is_published
    }
  }
}
